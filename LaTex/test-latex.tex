\text{The logarithmic relation for the near wall velocity is given by:}
\begin{subequations}
\setlength{\jot}{15pt}
    \begin{align}
        u^+ &= \frac{U_t}{u_\tau} = \frac{1}{K} \cdot \ln{y^+} + C\\
        y^+ &= \frac{\rho \cdot \Delta \cdot y \cdot u_\tau}{\mu} \\
        u_\tau &= (\frac{\tau_\omega}{\rho})^{\frac{1}{2}}
        \end{align}  
\end{subequations}
